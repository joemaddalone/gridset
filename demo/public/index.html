Gridset

Selection Movement Source

Setup

import { Gridset } from 'gridset';
const grid = new Gridset({
  width: 250, // <-- width of the grid
  height: 250, // <-- height of the grid
  rows: 5, // <-- number of rows
  cols: 5, // <-- number of columns
});
